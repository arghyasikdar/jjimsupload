# ═══════════════════════════════════════════════
#  SGD with Momentum (Nesterov optional)
# ═══════════════════════════════════════════════
class SGDMomentum:
    def __init__(self, lr=0.01, momentum=0.9,
                 weight_decay=1e-4, nesterov=False):
        self.lr           = lr
        self.momentum     = momentum
        self.weight_decay = weight_decay
        self.nesterov     = nesterov
        self.velocity     = {}   # v_t per parameter
        self.t            = 0

    def step(self, params, grads):
        self.t += 1
        for name, param in params.items():
            g = grads[name]

            # L2 regularisation (weight decay)
            if self.weight_decay > 0:
                g = g + self.weight_decay * param

            # Initialise velocity
            if name not in self.velocity:
                self.velocity[name] = np.zeros_like(param)

            v = self.velocity[name]

            # Momentum update:  v_t = β·v_{t-1} + g_t
            v_new = self.momentum * v + g
            self.velocity[name] = v_new

            if self.nesterov:
                # Nesterov: look-ahead gradient
                # θ ← θ − η·(β·v_new + g)
                param -= self.lr * (self.momentum * v_new + g)
            else:
                # Classic momentum:  θ ← θ − η·v_t
                param -= self.lr * v_new

# ─── Key intuition ──────────────────────────────────
#  The velocity term accumulates gradient history.
#  High momentum (β → 1) = longer memory = faster
#  convergence in consistent directions, but risks
#  overshooting in sharp minima.
#
#  Effective update rule visualised:
#    v_t  = β·v_{t-1} + (1-β)·g_t   (EMA of grads)
#    θ_t  = θ_{t-1} − η·v_t
#
#  Convergence: O(1/√T) for non-convex, O(1/T) convex
# ────────────────────────────────────────────────────

# LR scheduler (step decay)
class StepLR:
    def __init__(self, optimizer, step_size=30, gamma=0.1):
        self.opt       = optimizer
        self.step_size = step_size
        self.gamma     = gamma

    def step(self, epoch):
        if (epoch + 1) % self.step_size == 0:
            self.opt.lr *= self.gamma

# ═══════════════════════════════════════════════
#  Adam — Adaptive Moment Estimation
#  Kingma & Ba, 2015  (arXiv:1412.6980)
# ═══════════════════════════════════════════════
class Adam:
    def __init__(self, lr=1e-3, beta1=0.9, beta2=0.999,
                 eps=1e-8, weight_decay=0, amsgrad=False):
        self.lr           = lr
        self.beta1        = beta1        # 1st moment decay
        self.beta2        = beta2        # 2nd moment decay
        self.eps          = eps
        self.weight_decay = weight_decay
        self.amsgrad      = amsgrad
        self.m  = {}   # 1st moment (mean)
        self.v  = {}   # 2nd moment (uncentered variance)
        self.vh = {}   # max of v (AMSGrad)
        self.t  = 0    # step counter

    def step(self, params, grads):
        self.t += 1
        for name, param in params.items():
            g = grads[name]

            if self.weight_decay > 0:
                g = g + self.weight_decay * param

            # Init moments
            if name not in self.m:
                self.m[name]  = np.zeros_like(param)
                self.v[name]  = np.zeros_like(param)
                self.vh[name] = np.zeros_like(param)

            # 1st moment: m_t = β₁·m_{t-1} + (1-β₁)·g_t
            self.m[name] = (self.beta1 * self.m[name]
                            + (1 - self.beta1) * g)

            # 2nd moment: v_t = β₂·v_{t-1} + (1-β₂)·g_t²
            self.v[name] = (self.beta2 * self.v[name]
                            + (1 - self.beta2) * g ** 2)

            # Bias correction
            m_hat = self.m[name] / (1 - self.beta1 ** self.t)
            v_hat = self.v[name] / (1 - self.beta2 ** self.t)

            if self.amsgrad:
                self.vh[name] = np.maximum(self.vh[name], v_hat)
                denom = np.sqrt(self.vh[name]) + self.eps
            else:
                denom = np.sqrt(v_hat) + self.eps

            # θ_t = θ_{t-1} − η·m̂_t / (√v̂_t + ε)
            param -= self.lr * m_hat / denom

# ─── Why bias correction? ───────────────────────────
#  At t=1: m_1 = (1-β₁)·g₁ ≈ 0  (initialised at 0)
#  Without correction the moments are biased toward 0
#  in early steps → very small parameter updates.
#  Dividing by (1-β^t) counteracts this warm-up bias.
# ────────────────────────────────────────────────────

# ═══════════════════════════════════════════════
#  RMSprop — Root Mean Square Propagation
#  Tieleman & Hinton, Coursera 2012
# ═══════════════════════════════════════════════
class RMSprop:
    def __init__(self, lr=1e-2, alpha=0.99,
                 eps=1e-8, momentum=0, centered=False):
        self.lr       = lr
        self.alpha    = alpha      # decay for square average
        self.eps      = eps
        self.momentum = momentum
        self.centered = centered  # subtract grad mean (centred RMS)
        self.v  = {}   # EMA of squared gradients
        self.g  = {}   # EMA of gradients (centred mode)
        self.buf = {}  # momentum buffer

    def step(self, params, grads):
        for name, param in params.items():
            g = grads[name]

            if name not in self.v:
                self.v[name]   = np.zeros_like(param)
                self.g[name]   = np.zeros_like(param)
                self.buf[name] = np.zeros_like(param)

            # v_t = α·v_{t-1} + (1-α)·g_t²
            self.v[name] = (self.alpha * self.v[name]
                            + (1 - self.alpha) * g ** 2)

            if self.centered:
                # Centred RMSprop: subtract the squared mean
                self.g[name] = (self.alpha * self.g[name]
                                + (1 - self.alpha) * g)
                denom = np.sqrt(
                    self.v[name] - self.g[name]**2 + self.eps
                )
            else:
                denom = np.sqrt(self.v[name]) + self.eps

            if self.momentum > 0:
                # θ_t = θ_{t-1} − μ·buf + η·g/denom
                self.buf[name] = (self.momentum * self.buf[name]
                                  + self.lr * g / denom)
                param -= self.buf[name]
            else:
                # θ_t = θ_{t-1} − η·g / (√v_t + ε)
                param -= self.lr * g / denom

# ─── Comparison vs Adam ──────────────────────────────
#  RMSprop = Adam without bias correction or 1st moment
#  Great for: RNNs, online learning, non-stationary
#  objectives where recent gradients matter more.
#
#  Centred RMSprop normalises by variance of gradients
#  rather than their second moment — more stable for
#  some loss landscapes.
# ────────────────────────────────────────────────────

#  Dataset generators used in this lab
# ═══════════════════════════════════════════════
import numpy as np
from sklearn.datasets import make_moons, make_circles, make_blobs
from sklearn.preprocessing import StandardScaler

def get_dataset(name, n=1000, noise=0.2, seed=42):
    rng = np.random.RandomState(seed)

    if name == 'moons':
        # Two interleaved half-circles — classic non-linear
        X, y = make_moons(n, noise=noise, random_state=seed)

    elif name == 'circles':
        # Concentric rings — requires radial decision boundary
        X, y = make_circles(n, noise=noise, factor=0.4,
                              random_state=seed)

    elif name == 'xor':
        # Classic XOR — 4-quadrant alternating labels
        X = rng.randn(n, 2)
        y = (X[:, 0] * X[:, 1] > 0).astype(int)
        X += rng.randn(n, 2) * noise

    elif name == 'spiral':
        # 3-class Archimedean spiral — hard, multi-class
        k, N = 3, n // 3
        X, y = [], []
        for c in range(k):
            t = np.linspace(0, 1, N)
            r = t * 4
            theta = t * 3 * np.pi + (2 * np.pi * c / k)
            X.append(np.c_[r * np.cos(theta) + rng.randn(N)*noise,
                             r * np.sin(theta) + rng.randn(N)*noise])
            y.append(np.full(N, c))
        X, y = np.vstack(X), np.hstack(y)

    elif name == 'regression':
        # Noisy sine wave — 1D regression task
        X = rng.uniform(-np.pi, np.pi, (n, 1))
        y = np.sin(X.ravel()) + rng.randn(n) * noise

    elif name == 'blobs':
        # 4 Gaussian clusters — multi-class, easy baseline
        X, y = make_blobs(n, centers=4, cluster_std=1.2,
                            random_state=seed)

    else:
        raise ValueError(f"Unknown dataset: {name}")

    X = StandardScaler().fit_transform(X)
    return X, y

# ─── Data split & loader ─────────────────────────────
from sklearn.model_selection import train_test_split

def load_split(name, val_ratio=0.2):
    X, y = get_dataset(name)
    return train_test_split(X, y, test_size=val_ratio,
                              stratify=y if y.ndim == 1 else None,
                              random_state=42)

#  Multi-Layer Perceptron (pure NumPy)
# ═══════════════════════════════════════════════
import numpy as np

class Linear:
    def __init__(self, in_f, out_f):
        # He (Kaiming) initialisation
        scale = np.sqrt(2.0 / in_f)
        self.W = np.random.randn(in_f, out_f) * scale
        self.b = np.zeros(out_f)

    def forward(self, x):
        self.x = x
        return x @ self.W + self.b

    def backward(self, dout):
        self.dW = self.x.T @ dout
        self.db = dout.sum(axis=0)
        return dout @ self.W.T

class ReLU:
    def forward(self, x):
        self.mask = x > 0
        return x * self.mask

    def backward(self, dout):
        return dout * self.mask

class Dropout:
    def __init__(self, p=0.1):
        self.p = p
        self.training = True

    def forward(self, x):
        if not self.training:
            return x
        self.mask = (np.random.rand(*x.shape) > self.p) / (1 - self.p)
        return x * self.mask

    def backward(self, dout):
        return dout * self.mask

class MLP:
    def __init__(self, input_dim, hidden_layers, output_dim,
                 activation='relu', dropout=0.1):
        dims = [input_dim] + hidden_layers + [output_dim]
        self.layers = []
        for i in range(len(dims) - 1):
            self.layers.append(Linear(dims[i], dims[i+1]))
            if i < len(dims) - 2:  # no activation on output
                self.layers.append(ReLU())
                if dropout > 0:
                    self.layers.append(Dropout(dropout))

    def forward(self, x):
        for layer in self.layers: x = layer.forward(x)
        return x

    def backward(self, dout):
        for layer in reversed(self.layers): dout = layer.backward(dout)

# ─── Loss functions ──────────────────────────────────
def cross_entropy(logits, y):
    # Numerically stable softmax + NLL
    logits -= logits.max(axis=1, keepdims=True)
    probs   = np.exp(logits) / np.exp(logits).sum(1, keepdims=True)
    N       = y.shape[0]
    loss    = -np.log(probs[range(N), y] + 1e-9).mean()
    dlogits = probs.copy()
    dlogits[range(N), y] -= 1
    return loss, dlogits / N

def mse_loss(pred, y):
    diff = pred.ravel() - y
    return (diff**2).mean(), 2 * diff / len(y)
